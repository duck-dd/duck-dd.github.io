golang singleflight

2025-04-05 · 创建于: 2025-04-05 · 3 min · 565 words
 ·  CS
 ·  Golang , 2025

高并发场景下，有时我们很多并发goroutine内有逻辑重合，可能拉了数据库同一张表的数据，可能拉了同一个远程图片，这些数据通常是偏静态的，那么没有必要在每个goroutine内部都做这个操作，可能会拉高上游服务的负载，可能会影响本地的执行效率，可能会浪费本地的网卡资源等。

通常我们会依赖缓存来解决这个问题，这也确实是行之有效的；但是golang并发场景也给了我们更简便的实现方法：golang.org/x/sync/singleflight。

如下面的例子，doSth有2s的Sleep模拟执行过程；我们分了两组for+routine，中间有1s的Sleep，确保第一个发起doSth的index是可控的，然后我们看到输出实际doSth只执行了一次，所有routine共享了同一份结果。 PS：所有的shared返回值都是true，可能并不是大脑第一印象的第一个实际执行者是false，其他人是true

package main

import (
	"fmt"
	"sync"
	"sync/atomic"
	"time"
	"golang.org/x/sync/singleflight"
)

func doSth(index int) (any, error) {
	time.Sleep(2 * time.Second)
	return fmt.Sprintf("result_%d", index), nil
}

func main() {
	var g singleflight.Group
	var wg sync.WaitGroup
	var doCount int32 

	for i := 100; i < 101; i++ {
		wg.Add(1)
		index := i
		go func() {
			defer wg.Done()
			ret, err, shared := g.Do("key", func() (interface{}, error) {
				atomic.AddInt32(&doCount, 1)
				v,e := doSth(index)
				fmt.Printf("index:%d, v: %v, err: %v\n", index, v, e)
				return v,e
			})
			fmt.Printf("index:%d, ret:%v, shared:%v, err:%v\n", index, ret, shared, err)
		}()
	}
	
	
	time.Sleep(time.Second)
	
	for i := 0; i < 5; i++ {
		wg.Add(1)
		index := i
		go func() {
			defer wg.Done()
			ret, err, shared := g.Do("key", func() (interface{}, error) {
				atomic.AddInt32(&doCount, 1)
				v,e := doSth(index)
				fmt.Printf("index:%d, v: %v, err: %v\n", index, v, e)
				return v,e
			})
			fmt.Printf("index:%d, ret:%v, shared:%v, err:%v\n", index, ret, shared, err)
		}()
	}
	
	wg.Wait()
	
	fmt.Printf("实际执行次数: %d\n", doCount)
}

真正发起共享的函数Do入参是有key的，我们把key做个替换，会发现不同key之间不共享，如下：

				return v,e
			})
			fmt.Printf("index:%d, ret:%v, shared:%v, err:%v\n", index, ret, shared, err)

}

基于缓存的思想，那么就会问了，这个被“缓存”的结果生命周期是多久呢？我们能不能设置每个key的缓存时间呢？答案是否定的，缓存时间不支持设置，他的生命周期是发起实际执行动作的函数没有执行完成期间所有加入这个key的调用会等待这份共享结果，如下例，第一轮for循环Do2s执行完成，即使routine还在，但是sleep 3s后下一轮for+routine执行时已经不再共享这份数据。

package main

import (
	"fmt"
	"sync"
	"sync/atomic"
	"time"
	"golang.org/x/sync/singleflight"
)

func doSth(index int) (any, error) {
	time.Sleep(2 * time.Second)
	return fmt.Sprintf("result_%d", index), nil
}

func main() {
	var g singleflight.Group
	var wg sync.WaitGroup
	var doCount int32 

	for i := 100; i < 101; i++ {
		wg.Add(1)
		index := i
		go func() {
			defer wg.Done()
			ret, err, shared := g.Do("key", func() (interface{}, error) {
				atomic.AddInt32(&doCount, 1)
				v,e := doSth(index)
				fmt.Printf("index:%d, v: %v, err: %v\n", index, v, e)
				return v,e
			})
			time.Sleep(10 * time.Second)
			fmt.Printf("index:%d, ret:%v, shared:%v, err:%v\n", index, ret, shared, err)
		}()
	}
	
	time.Sleep(3 * time.Second)
	
	for i := 0; i < 5; i++ {
		wg.Add(1)
		index := i
		go func() {
			defer wg.Done()
			ret, err, shared := g.Do("key", func() (interface{}, error) {
				atomic.AddInt32(&doCount, 1)
				v,e := doSth(index)
				fmt.Printf("index:%d, v: %v, err: %v\n", index, v, e)
				return v,e
			})
			fmt.Printf("index:%d, ret:%v, shared:%v, err:%v\n", index, ret, shared, err)
		}()
	}
	
	wg.Wait()
	
	fmt.Printf("实际执行次数: %d\n", doCount)
}
